## 批量生成脚本示例

```bash
#!/bin/bash

# 模块列表
modules=(
  "4:告警与响应"
  "5:分布式追踪与诊断"
  "6:可视化与报告"
  "7:安全与访问控制"
  "8:合规与审计"
  "9:高可用与灾备"
  "10:性能与扩展"
  "11:自动化运维"
  "12:API与集成"
  "13:用户体验"
  "14:协作与工作流"
  "15:企业级特性"
  "16:高级功能"
  "17:备份系统增强"
  "18:真实备份集成"
  "19:通用日志采集代理"
  "20:ML/AI机器学习框架"
  "21:NLP自然语言处理"
  "22:多租户架构"
  "23:边缘计算"
  "24:成本管理"
  "25:数据模型与系统接口"
)

# 遍历生成
for module in "${modules[@]}"; do
  IFS=':' read -r num name <<< "$module"
  
  # 复制模板
  cp design-module1.md "design-module${num}.md"
  
  # 替换模块信息
  sed -i "s/模块一：日志采集/模块${num}：${name}/g" "design-module${num}.md"
  sed -i "s/requirements-module1.md/requirements-module${num}.md/g" "design-module${num}.md"
  
  echo "已生成 design-module${num}.md"
done
```
